```c++
class NonCopyable {
public:
    NonCopyable() = default;
    
    // Запрет копирования
    NonCopyable(const NonCopyable&) = delete;
    NonCopyable& operator = (const NonCopyable&) = delete;
    
    // Запрет перемещения
    NonCopyable(NonCopyable&&) = delete;
    NonCopyable& operator = (NonCopyable&&) = delete;
};
```
