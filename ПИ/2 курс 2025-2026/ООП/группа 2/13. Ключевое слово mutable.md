Применяется к полям класса и позволяет изменять их даже в константных методах:
``` c++
class Cache {
    mutable bool cacheValid{false};
    mutable int cachedValue;
public:
    int getValue() const {
        if (!cacheValid) {
            cachedValue = computeValue(); // Разрешено, т.к. mutable
            cacheValid = true;
        }
        return cachedValue;
    }
};
```
